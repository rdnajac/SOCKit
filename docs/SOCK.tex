% SOCKit language definition for listings
\lstdefinelanguage{SOCKit}{
    keywords={true, false, bool}, % Keywords

    morecomment=[l]{//}, % Line comments
    morecomment=[s]{/*}{*/}, % Block comments

    morestring=[b]', % Single-quoted strings
    morestring=[b]", % Double-quoted strings

    keywordstyle=\color{blue},
    commentstyle=\color{green},
    stringstyle=\color{red},

    % Basic styling for code listing
    basicstyle=\ttfamily\footnotesize,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    numbers=left,                    % Line numbers on the left
    numberstyle=\tiny\color{gray},   % Ensure that the color package is included in your main document.
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
}

% End of SOCKit language definition
